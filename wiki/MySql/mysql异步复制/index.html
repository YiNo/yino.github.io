    Mysql异步复制

    #master

    配置文件开启(master)
server-id               = 1
log_bin                 = /var/log/mysql/mysql-bin.log
expire_logs_days        = 10
max_binlog_size   = 100M

#创建用户 并赋予复制权限（master）
create user 'test'@'192.168.0.%' identified by '';
grant replication slave on *.* to 'test'@'192.168.0.%';
flush privileges; #刷新权限
show master status; #查看master状态 binlog日志，position
导出master 数据
mysqldump --single-transaction --master-data=2 --triggers --routines --all-databases -uroot -p > all.sql

    #slave

    将all.sql 复制到slave 服务上 scp all.sql user@ip:/var/log/mysql
导入 master 数据 mysql -uroot -p< all.sql
修改与master的链接
change master to master_host = '192.168.1.1200',
master_user='dba'，
master_password='123456',
master_log_file = 'master-bin.000002', #来自于show master status; 中的bin-log字段
master_log_pos=154; #来自于show master status; 中的postition字段
#查看slave status 状态
Start slave;（开启）;开启
show slave status \G;(查看主从进程连接状态) Slave_IO_Running，Slave_SQL_Running  配置成功后 这俩状态是YES
